# grok.nvim Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Neovim plugin that runs Grok Build as an ACP sidebar client so you can request work, review diffs (or auto-apply), and watch code change without leaving the editor.

**Architecture:** Thin Lua ACP client over `grok agent stdio` (JSON-RPC ndjson). Sidebar buffer for chat/permissions; main area for code and a single-file review diff. Review mode gates edits via `session/request_permission`; Auto mode auto-allows and follows tool locations.

**Tech Stack:** Neovim ≥ 0.10 Lua, `vim.system`/`jobstart` for the agent process, plenary.nvim for tests (devDependency only), zero required runtime UI plugins.

**Spec:** `docs/superpowers/specs/2026-07-13-grok-nvim-design.md`

## Global Constraints

- Do **not** `git init` or `git commit` unless the user explicitly asks. Skip any commit steps by default.
- No hard dependency on snacks.nvim, nui, or telescope. Optional picker only if present.
- Do not reverse-engineer Claude Code’s lockfile/WebSocket IDE protocol.
- One active review diff at a time; queue additional edit permissions FIFO.
- Default `permission_mode = "review"`.
- Keymap prefix defaults to `<leader>g` (avoid `<leader>a` used by claudecode).
- Pure Lua; no Node/Python runtime deps for the plugin itself.
- Prefer small modules; no god files.

`lua/grok/config.lua`:

```lua
local M = {}

local defaults = {
  cmd = { "grok", "agent", "stdio" },
  model = nil,
  cwd = nil,
  permission_mode = "review",
  sidebar = { position = "right", width = 0.36 },
  thoughts = "collapsed",
  follow = { enabled = true },
}

local current = vim.deepcopy(defaults)

local function deep_merge(dst, src)
  for k, v in pairs(src) do
    if type(v) == "table" and type(dst[k]) == "table" and not vim.islist(v) then
      deep_merge(dst[k], v)
    else
      dst[k] = v
    end
  end
  return dst
end

function M.reset()
  current = vim.deepcopy(defaults)
end

function M.setup(opts)
  opts = opts or {}
  local next_cfg = deep_merge(vim.deepcopy(defaults), opts)
  if next_cfg.permission_mode ~= "review" and next_cfg.permission_mode ~= "auto" then
    error("grok.nvim: permission_mode must be 'review' or 'auto'")
  end
  if next_cfg.thoughts ~= "collapsed" and next_cfg.thoughts ~= "expanded" and next_cfg.thoughts ~= "hidden" then
    error("grok.nvim: thoughts must be collapsed|expanded|hidden")
  end
  current = next_cfg
  return current
end

function M.get()
  return current
end

return M
```

```lua
local config = require("grok.config")
local M = {}
local current = "review"

function M.sync_from_config()
  current = config.get().permission_mode
end

function M.get()
  return current
end

function M.is_auto()
  return current == "auto"
end

function M.set(m)
  if m ~= "review" and m ~= "auto" then
    error("invalid mode: " .. tostring(m))
  end
  current = m
  return current
end

function M.toggle()
  return M.set(current == "auto" and "review" or "auto")
end

function M.pick_permission_option(options)
  if current ~= "auto" then
    return nil
  end
  local fallback = nil
  for _, opt in ipairs(options or {}) do
    if opt.kind == "allow_once" then
      return opt.optionId
    end
    if opt.kind == "allow_always" and not fallback then
      fallback = opt.optionId
    end
  end
  return fallback
end

return M
```

**Interfaces:**
- Produces: `Client.new(opts) -> client` where `opts = { cmd=string[], cwd=string?, on_notification=fn(msg), on_exit=fn(code) }`
- Produces: `client:start() -> nil` (spawns process; for tests can inject `transport`)
- Produces: `client:request(method, params) -> promise-like` via callback or coroutine: actually use **callback style** `client:request(method, params, cb)` where `cb(err, result)`
- Produces: `client:notify(method, params)`
- Produces: `client:respond(id, result)` for answering server→client requests (permissions)
- Produces: `client:stop()`
- Internal: parse stdout lines as JSON; route by `id` to pending requests; if `method` without matching client request, treat as server request/notification and call `on_notification` / `on_server_request`
- For tests: `Client.new({ transport = mock_transport })` where transport has `write(line)`, and test drives `client:_on_stdout_line(line)`

**Important protocol shapes (Grok):**

Outbound request:

```json
{"jsonrpc":"2.0","id":1,"method":"initialize","params":{...}}
```

Inbound result:

```json
{"jsonrpc":"2.0","id":1,"result":{...}}
```

Inbound notification:

```json
{"jsonrpc":"2.0","method":"session/update","params":{...}}
```

Inbound server request (permissions):

```json
{"jsonrpc":"2.0","id":5,"method":"session/request_permission","params":{...}}
```

Core of `lua/grok/acp/client.lua`:

```lua
local M = {}
local Client = {}
Client.__index = Client

function M.new(opts)
  opts = opts or {}
  local self = setmetatable({
    cmd = opts.cmd or { "grok", "agent", "stdio" },
    cwd = opts.cwd,
    on_notification = opts.on_notification or function() end,
    on_server_request = opts.on_server_request or function() end,
    on_exit = opts.on_exit or function() end,
    _next_id = 1,
    _pending = {},
    _job = nil,
    _buf = "",
    _write_fn = opts._write,
    transport = opts.transport,
  }, Client)
  return self
end

function Client:_write(line)
  if self._write_fn then
    self._write_fn(line)
    return
  end
  if self._job then
    -- jobstart channel write; implement with vim.fn.chansend
    vim.fn.chansend(self._job, line .. "\n")
  end
end

function Client:request(method, params, cb)
  local id = self._next_id
  self._next_id = id + 1
  self._pending[id] = cb
  self:_write(vim.json.encode({
    jsonrpc = "2.0",
    id = id,
    method = method,
    params = params or {},
  }))
  return id
end

function Client:notify(method, params)
  self:_write(vim.json.encode({
    jsonrpc = "2.0",
    method = method,
    params = params or {},
  }))
end

function Client:respond(id, result)
  self:_write(vim.json.encode({
    jsonrpc = "2.0",
    id = id,
    result = result,
  }))
end

function Client:_on_stdout_line(line)
  if line == nil or line == "" then
    return
  end
  local ok, msg = pcall(vim.json.decode, line)
  if not ok then
    return -- or log
  end
  if msg.id ~= nil and (msg.result ~= nil or msg.error ~= nil) and not msg.method then
    local cb = self._pending[msg.id]
    self._pending[msg.id] = nil
    if cb then
      if msg.error then
        cb(msg.error, nil)
      else
        cb(nil, msg.result)
      end
    end
    return
  end
  if msg.method and msg.id ~= nil then
    self.on_server_request(msg)
    return
  end
  if msg.method then
    self.on_notification(msg)
  end
end

function Client:start()
  if self.transport == "manual" then
    return
  end
  local cfg = {
    cwd = self.cwd,
    stdin = "pipe",
    stdout = function(_, data)
      if not data then
        return
      end
      -- jobstart may pass array of lines; normalize
      local chunk = type(data) == "table" and table.concat(data, "\n") or data
      self._buf = self._buf .. chunk
      while true do
        local idx = self._buf:find("\n", 1, true)
        if not idx then
          break
        end
        local line = self._buf:sub(1, idx - 1)
        self._buf = self._buf:sub(idx + 1)
        vim.schedule(function()
          self:_on_stdout_line(line)
        end)
      end
    end,
    stderr = function() end,
    on_exit = function(_, code)
      vim.schedule(function()
        self.on_exit(code)
      end)
    end,
  }
  -- Prefer vim.system if streaming stdin is awkward; use jobstart for bidirectional:
  self._job = vim.fn.jobstart(self.cmd, {
    cwd = self.cwd,
    rpc = false,
    stdout_buffered = false,
    stderr_buffered = false,
    on_stdout = function(_, data, _)
      if not data then
        return
      end
      for _, line in ipairs(data) do
        if line ~= "" then
          vim.schedule(function()
            self:_on_stdout_line(line)
          end)
        end
      end
    end,
    on_stderr = function() end,
    on_exit = function(_, code)
      vim.schedule(function()
        self.on_exit(code)
      end)
    end,
  })
  if self._job <= 0 then
    error("failed to start grok agent")
  end
end

function Client:stop()
  if self._job and self._job > 0 then
    vim.fn.jobstop(self._job)
    self._job = nil
  end
end

return M
```

Orchestration sketch in `init.lua`:

```lua
function M.send(text)
  if not text or text == "" then return end
  local cfg = config.get()
  ensure_client()
  session.ensure(client, cfg.cwd or vim.fn.getcwd(), function(err)
    if err then render_error(err) return end
    render.append_user(chat_buf, text)
    session.set_busy(true)
    client:request("session/prompt", protocol.prompt_params(session.get_id(), {
      { type = "text", text = text },
    }), function(req_err, result)
      session.set_busy(false)
      if req_err then render_error(req_err) end
      -- result.stopReason available
    end)
  end)
end
```
